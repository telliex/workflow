# (260)分解条件表达式 / Decompose Conditional

简式

if(!sDate.isBefore(plan.summerStart) && !aDate.isAfter(plan.summerEnd))
  charge=quantity * plan.summerRate;
else
  charge=quantity * plan.regularRate + plan.regularServiceCharge;

⇩

if(summer())
  charge=summerCharge();
else
  charge=regularCharge();

动机

  1. 程序中，複杂的条件最常导致複杂度上升。
  2. 条件判断式裡的大块代码，可以分解为许多独立的函数，根据每个小块代码的用途，为分解而得的新函数命名。
  3. 将每个分支条件分解成新函数可以突出条件逻辑，更清楚的表明每个分支的作用。

# 完整范例

if(!sDate.isBefore(plan.summerStart) && !aDate.isAfter(plan.summerEnd)){
  charge=quantity * plan.summerRate;
}else{
  charge=quantity * plan.regularRate + plan.regularServiceCharge; 
}

⇩


charge= summer()? summerCharge() : regularCharge();

function summer(){
  return !sDate.isBefore(plan.summerStart) && !aDate.isAfter(plan.summerEnd);
}

function summerCharge(){
  return quantity * plan.summerRate;
}

function regularCharge(){
  return quantity * plan.regularRate + plan.regularServiceCharge;
}

# (263)合并条件表达式 / Consolidate Conditional Expression

简式

if(anEmployee.seniorty < 2) return 0;
if(anEmployee.monthsDisabled > 12) return 0;
if(anEmployee.isPartTime) return 0;

⇩

if(isNotEligibleForDisability()) return 0;

function isNotEligibleForDisability(){
  return ((anEmployee.seniority <2) 
          || (anEmployee.monthsDisabled > 12) 
          || (anEmployee.isPartTime));
}

动机

  1. 当判断式裡的检查条件不相同，最终行为却是一致，我会考虑将其合併，如此仅需一次检查。
  2. 如果这些判断式确实彼此独立，不应该视为同一次检查，就不该使用本重构。
  3. 合并后的代码混乱，可适时的使用提炼函数(106)。

# (266)以卫语句取代嵌套条件表达式 / Replace Nested Conditional with Guard Clauses

简式

function getPlayAmount(){
  let result;
  if(isDead){
    result=deadAmount();
  }else{
    if(isSeparated){
      result=separatedAmount();
    }else{
      if(isRetired){
        result=retiredAmount();
      }else{
        result=normalPayAmount();
      }
    }
  }
  return result;
}

⇩

function getPayAmount(){
  if(isDead) return deadAmount();
  if(isSeparated) return separatedAmount();
  if(isRetired) return retiredAmount();
  return normalPayAmount();
}

动机

  1. 如果某个条件极其罕见，就应该单独检查该条件，并在条件为真时从函数中返回。这样的检查称为卫语句(guard clauses)
  2. 卫语句的使用，作用在告诉阅读者"这情况不是本函数核心逻辑所关心的，如果发生了，请做一些必要的处理，然后退出"。

# (272)以多态取代条件表达式 / Replace Conditional with Polymorphism

简式

switch(bird.type){
  case 'EuropeanSwallow':
    return "average";
  case 'AfricanSwallow':
    return (bird.numberOfCoconuts >2) ? 'tired' : 'average';
  case 'NorwegianBlueParrot':
    return (bird.voltage > 100)? 'scorched':'beautiful';
  default:
    return 'unknown';
}

⇩

class EuropeanSwallow{
  get plumage(){
    return 'average';
  }
}
class AfricanSwallow{
  get plumage(){
    return (this.numberOfCoconuts > 2 )? 'tired' : 'average';
  }
}

class NorwegianBlueParrot{
  get plumage(){
    return (this.voltage >100)? 'scorched' : 'beautiful';
  }
}

动机

  1. 仅是简单的条件语句，是不需劳师动众引入多态。但若是复杂的的条件逻辑，多态是改善复杂情况的有力工具。
  2. 常见的场景是，创建一组类型，每个类型处理各自的一种条件逻辑。
  3. 另一种情况是，有一基础逻辑放进超类，然后将每种变体逻辑单独放进一个子类。

# (289)引入特例 / Introduce Special Case

简式

if(aCustomer==='unknow') customerName='occupant';

⇩

class UnknownCustomer{
  get name() {
    return 'occupant';
  }
}

动机

  1. 特例模式,创建一个特例元素，用以表达对这种特例的共用行为。可用一个函数掉用取代大部分特例检查。
  2. 一个常需要特例处理的值是 null ，这模式也常被称为null 对象模式。

# 完整范例

创建一个类

class Site{
  constructor(){
    this._customer=new Customer();
  }
  get customer(){
    return this._customer;
  }
}
class Customer{
  get name(){...};
  get billingPlan(){...};
  set billingPlan(arg){...};
  get paymentHistory(){...};
}

// 客户端 1
const site=new Site();
const aCustomer=site.customer
let customerName;
if(aCustomer==='unknown') customerName='occupant';
else customerName=aCustomer.name;

// 客户端 2
const plan=(aCustomer==='unknown')?registry.billingPlans.basic : aCustomer.billingPlan;

// 客户端 3
if(aCustomer!=='unknown') aCustomer.billingPlan=newPlan;

// 客户端 4
const weeksDelinquent=(aCustomer==='unknown')? 0 : aCustomer.paymentHistory.weeksDelinquentInLastYear;

⇩

class Site{
  constructor(){
    this._customer=new Customer();
  }
  get customer(){
    // return this._customer;
    return (this._customer==='unknown')? new UnknownCustomer() : this._customer;
  }
}
class Customer{
  get isUnknown(){
    return false;
  }
  get name(){...};
  get billingPlan(){...};
  set billingPlan(arg){...};
  get paymentHistory(){...};
}

class UnknownCustomer{
  get isUnknown(){
    return true;
  }
  get name(){
    return 'occupant';
  }
  get billingPlan(){ return registry.billingPlans.basic };
  set billingPlan(arg){...};
  get paymentHistory(){ return new NullPaymentHistory(); }
}

class NullPaymentHistory{
  get weeksDelinquentInLastYear(){ return 0 ;}
}


// 客户端 1

使用对象

class Site{
  constructor(){
    this._customer=new Customer();
  }
  get customer(){
    // return this._customer;
    return (this._customer==='unknown')? createUnknownCustomer() : this._customer;
  }

// 顶层作用域可建立此 function 当成特例

function createUnknownCustomer(){
  return {
    isUnknown:true,
    name:'occupant',
    billingPlan:registry.billingPlans.basic,
    paymentHistory:{

    }
  }
}

# (302)引入断言 / Introduce Assertion

简式

if(this.discountRate){
  base=base-(this.discountRate*base);
}

⇩

assert(this.discountRate >=0);
if(this.discountRate){
  base=base-(this.discountRate*base);
}

动机

  1. 断言是一种很有价值的交流形式 - 他告诉阅读者，程序在执行到此处时，对当前状态做了何种假设。
  2. 使用断言比注解更容易传达讯息。
  3. 我不会用断言来检查所有我认为应该为真的条件，只用来检查必须为真的条件。
